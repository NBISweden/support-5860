2 Overview

Healthy-diseased dataset with both conditions are used for Trajectory analyses. Trajectories were created using Slingshot and Scorpius. Genes associated with the trajectory are identified through function testPseudotime() from TSCAN package. These steps are repeated on 10 cluster grouping, 9 cluster grouping and 7 cluster grouping.

# custom ggplot theme
theme_report <- function (basesize=12,lpos="top",basefamily="Barlow") {
    theme_bw(base_size=basesize,base_family=basefamily) %+replace%
        theme(
            panel.border=element_blank(),
            panel.grid.minor=element_blank(),
            panel.grid.major.x=element_blank(),
            legend.position=lpos,
            strip.background=element_blank(),
            axis.ticks.y=element_blank(),
            axis.ticks.x=element_line(colour="grey60"),
        plot.caption=element_text(hjust=0,colour="grey60",size=10),
        plot.title=element_text(colour="grey60",hjust=0),
        plot.subtitle=element_text(colour="grey60")
        )
}

# custom ggplot theme
theme_simple <- function (basesize=12,lpos="top",basefamily="Barlow") {

            panel.border=element_blank(),

            legend.position=lpos,
            strip.background=element_blank(),
            axis.ticks=element_blank(),
            axis.text=element_blank(),

        plot.caption=element_text(hjust=0,colour="grey60",size=10),
        plot.title=element_text(colour="grey60"),
        plot.subtitle=element_text(colour="grey60")
        )
}
