1 Overview

Healthy dataset with both conditions are used for Trajectory analyses. Trajectories were created using Scorpius. Genes associated with the trajectory are identified through function testPseudotime() from TSCAN package. These steps are run on full dataset, subset without clusters 6 and 7, and clusters without 6, 7 and 8.

# custom ggplot theme
theme_report <- function (basesize=12,lpos="top",basefamily="Barlow") {
    theme_bw(base_size=basesize,base_family=basefamily) %+replace%
        theme(
            panel.border=element_blank(),
            panel.grid.minor=element_blank(),
            panel.grid.major.x=element_blank(),
            legend.position=lpos,
            strip.background=element_blank(),
            axis.ticks.y=element_blank(),
            axis.ticks.x=element_line(colour="grey60"),
        plot.caption=element_text(hjust=0,colour="grey60",size=10),
        plot.title=element_text(colour="grey60",hjust=0),
        plot.subtitle=element_text(colour="grey60")
        )
}

# custom ggplot theme
theme_simple <- function (basesize=12,lpos="top",basefamily="Barlow") {
    theme_bw(base_size=basesize,base_family=basefamily) %+replace%
        theme(
            panel.border=element_blank(),
            panel.grid=element_blank(),
            legend.position=lpos,
            strip.background=element_blank(),
            axis.ticks=element_blank(),
            axis.text=element_blank(),
            axis.title=element_blank(),
        plot.caption=element_text(hjust=0,colour="grey60",size=10),
        plot.title=element_text(colour="grey60"),
        plot.subtitle=element_text(colour="grey60")
        )
}

lec <- readRDS(file.path(path_export,"11012022_healthy_SCT_FINAL.rds"))

# create colours
lec_cols <- randomcoloR::distinctColorPalette(8)

saveRDS(lec_cols,file.path(path_export,"8-clusters/colours_8.Rds"))
library(SCORPIUS)
expression <- t(as.matrix(lec@assays$SCT@data))
groups <- lec$seurat_clusters
space <- as.matrix(Embeddings(lec,"umap"))
traj <- infer_trajectory(space,k=8)

#draw_trajectory_plot(space, progression_group=groups, contour=TRUE,progression_group_palette = lec_cols,point_size=0.8)

#draw_trajectory_plot(

#  progression_group = groups,
#  path = traj$path,
#  contour = TRUE,
#  point_size=0.8
#)
gg <- UMAPPlot(lec,cols=lec_cols)+
  geom_path(data=data.frame(traj$path), aes(x=Comp1, y=Comp2), size=1, alpha=0.7)+
  theme_report()+
  coord_fixed()
ggsave(file.path(path_export,"8-clusters/lec-scorpius-scatter.png"),gg,height=5.2,width=5)

Fig. 1: UMAP scatterplot of LEC subset showing clusters and scorpius path.

#gimp <- gene_importances(expression, traj$time, num_permutations=0, num_threads=10)
#expr_sel <- expression[,gimp[1:50,]$gene]
#ph <- draw_trajectory_heatmap(expr_sel, traj$time, groups, show_labels_row=T)
#ggsave(file.path(path_export,"8-clusters/lec-scorpius-heatmap.png"),ph,height=8,width=8)
# create sce object and add clusters
lec_sce <- Seurat::as.SingleCellExperiment(lec)
colData(lec_sce)$clusters <- colData(lec_sce)$seurat_clusters
reducedDim(lec_sce, "UMAP") <- Seurat::Embeddings(lec,reduction="umap")
lec_sce$scorpius <- traj$time

pseudo <- TSCAN::testPseudotime(lec_sce, pseudotime=lec_sce$scorpius) %>%
            as.data.frame() %>%
            tibble::rownames_to_column("gene") %>%
            filter(FDR<0.05) %>%
            arrange(FDR)

ph <- scater::plotHeatmap(lec_sce, order_columns_by="scorpius",
    colour_columns_by="clusters", features=pseudo$gene[1:40],
    column_annotation_colors = list("clusters"=lec_cols[names(lec_cols) %in% unique(lec_sce$clusters)]),color=viridis(8,option="cividis"))
ggsave(file.path(path_export,"8-clusters/lec-scorpius-heatmap.png"),ph,height=8,width=8)
